125 void foreach_setup_in_group(ISpecSetup* setup) {
126 if (_currentSpecFailed) {
127 auto result = SpecRunResult::not_run(setup, _currentSpec);
128 _reporters->report_setup(result.get());
129 return;

131
132 _currentRunResultPromise = std::make_unique<std::promise<ISpecRunResult*>>();

134 auto* codeBlock = setup->code_block();
135 if (codeBlock->get_timeout_ms() == 0) codeBlock->set_timeout_ms(_timeoutMs);
136
137 codeBlock->run(setup, _currentSpec, &_codeBlockCallbackFn);
138
139 auto future = _currentRunResultPromise->get_future();
140 if (codeBlock->get_timeout_ms() > 0) {
141 if (future.wait_for(std::chrono::milliseconds(codeBlock->get_timeout_ms())) ==
142 std::future_status::timeout) {
143 auto result = SpecRunResult::timeout(setup, _currentSpec);
144 _reporters->report_setup(result.get());
145 _currentSpecFailed = true;
146 return;
147 }
148 }
149
150 if (auto* result = future.get()) _reporters->report_setup(result);

185 void foreach_spec_in_group(ISpec* spec) {
186 if (_currentlySkippingTests || list_only() || !should_run_spec(spec)) {
187 auto result = SpecRunResult::not_run(spec, spec);
188 _reporters->report_spec_result(result.get());
189 _resultTotalCounts.increment_not_run();
190 return;
191 }
192
193 _currentSpec = spec;
194 _currentSpecFailed = false;
195 _currentResult = nullptr;
196
197 _reporters->report_spec_begin(_currentSpec);
198
199 std::vector<ISpecGroup*> groupStack;
200 auto* currentGroup = spec->group();
201 while (currentGroup) {
202 groupStack.push_back(currentGroup);
203 currentGroup = currentGroup->group();
204 }
205
206 // Run foreach blocks, starting with root parent
207 for (auto it = groupStack.rbegin(); it != groupStack.rend(); ++it) {
208 auto* group = *it;
209 group->foreach_setup(&_forEachSetupInGroupFn);
210 if (!_currentSpecFailed) _setupGroupsRun++;
211 }
212
213 if (_currentSpecFailed) {
214 auto specCodeResult = SpecRunResult::not_run(spec, spec);
215 _reporters->report_spec(specCodeResult.get());
216
217 if (_currentResult && _currentResult->status() == RunResultStatus::Timeout) {
218 auto specFinalResult = SpecRunResult::timeout(spec, spec);
219 _reporters->report_spec_result(specFinalResult.get());
220 } else {
221 auto specFinalResult =
222 SpecRunResult::failed(spec, spec, _currentSpecFailureMessage);
223 _reporters->report_spec_result(specFinalResult.get());
224 }
225
226 _resultTotalCounts.increment_failed();
227
228 // We still tear things down - but only run the teardown for each group that run
229 // setups
230 int teardownsToRun = _setupGroupsRun;
231 auto groupCount = groupStack.size();
232 auto groupsToSkip = groupCount - teardownsToRun;
233
234 // Run teardowns, but skip the first X ones, where X is the number of groups
235 // that ran setups
236 for (int i = 0; i < groupCount; i++) {
237 if (i < groupsToSkip - 1) continue;
238 auto* group = groupStack[i];
239 group->foreach_teardown(&_forEachTeardownInGroupFn);
240 }
241
242 spec->variables()->delete_all();
243 return;
244 }
245
246 _currentRunResultPromise = std::make_unique<std::promise<ISpecRunResult*>>();
247
248 auto* codeBlock = spec->code_block();
249 if (codeBlock->get_timeout_ms() == 0) codeBlock->set_timeout_ms(_timeoutMs);
250
251 codeBlock->run(spec, spec, &_codeBlockCallbackFn);
252
253 auto future = _currentRunResultPromise->get_future();
254
255 if (codeBlock->get_timeout_ms() > 0) {
256 if (future.wait_for(std::chrono::milliseconds(codeBlock->get_timeout_ms())) ==
257 std::future_status::timeout) {
258 auto result = SpecRunResult::timeout(spec, spec);
259 _reporters->report_spec(result.get());
260 auto specFinalResult = SpecRunResult::failed(spec, spec);
261 _reporters->report_spec_result(specFinalResult.get());
262 _resultTotalCounts.increment_failed();
263 spec->variables()->delete_all();
264 return;
265 }
266 }
267
268 if (auto* result = future.get()) _reporters->report_spec(result);
269 else _Log_("Spec callback future.get() returned nullptr");
270
271 // Run teardown blocks, starting with current and walking up to root parent
272 for (auto* group : groupStack) group->foreach_teardown(&_forEachTeardownInGroupFn);
273
274 if (_currentSpecFailed) {
275 _resultTotalCounts.increment_failed();
276 auto specFinalResult =
277 SpecRunResult::failed(spec, spec, _currentSpecFailureMessage);
278 _reporters->report_spec_result(specFinalResult.get());
279 } else {
280 _resultTotalCounts.increment_passed();
281 auto specFinalResult = SpecRunResult::passed(spec, spec);
282 _reporters->report_spec_result(specFinalResult.get());
283 }
284 spec->variables()->delete_all();
285 }

291 void foreach_group_in_group(ISpecGroup* group) {
292 // TODO group these into an inline function:
293 _currentSpec = nullptr;
294 _currentSpecFailed = false;
295 _currentSpecFailureMessage.clear();
296 _currentResult = nullptr;
297
298 if (_currentlySkippingTests) {
299 group->foreach_spec(&_forEachSpecInGroupFn);
300 group->foreach_group(&_forEachGroupInGroupFn);
301 return;
302 }
303
304 bool shouldRun = should_run_group(group);
305 _currentlySkippingTests = !shouldRun;
306
307 // Run the group's one time setups
308 if (!_currentlySkippingTests)
309 group->foreach_one_time_setup(&_forEachSetupInGroupFn);
310
311 group->foreach_spec(&_forEachSpecInGroupFn);
312 group->foreach_group(&_forEachGroupInGroupFn);
313
314 // Run the group's one time teardowns
315 if (!_currentlySkippingTests)
316 group->foreach_one_time_teardown(&_forEachTeardownInGroupFn);
317
318 group->variables()->delete_all();
319
320 _currentlySkippingTests = false;
