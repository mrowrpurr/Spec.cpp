193 void run(ISpecComponent* self, ISpec* spec, ISpecRunResultCallbackFn* resultCallback)
194 override {
195 try {
196 if (is_async()) {
197 _asyncDonePromise = std::make_unique<std::promise<void>>();
198 body()->invoke(self, spec, &_markAsyncPromiseDoneFn);
199
200 auto asyncDoneFuture = _asyncDonePromise->get_future();
201 if (get_timeout_ms() > 0) {
202 if (asyncDoneFuture.wait_for(std::chrono::milliseconds(get_timeout_ms())) ==
203 std::future_status::timeout) {
204 if (resultCallback) {
205 auto result = SpecRunResult::timeout(self, spec);
206 resultCallback->invoke(result.get());
207 return;
208 } else {
209 _Log_("Code block timed out");
210 }
211 }
212 }
213
214 asyncDoneFuture.get();
215
216 } else {
217 body()->invoke(self, spec, nullptr);
218 }
219
220 if (!resultCallback) return;
221 auto result = SpecRunResult::passed(self, spec);
222 resultCallback->invoke(result.get());
223 } catch (...) {
224 _exceptionHandled = false;
225
226 // ...
227 if (_asyncDonePromise) _asyncDonePromise->set_exception(std::current_exception());
228
229 if (auto* currentlyRunningSpecEnvironment = global_spec_environment().get()) {
230 if (auto* exceptionHandlers =
231 currentlyRunningSpecEnvironment->local_exception_handlers()) {
232 _currentExceptionMessage.clear();
233 _currentExceptionPtr =
234 std::make_unique<std::exception_ptr>(std::current_exception());
235 exceptionHandlers->foreach_exception_handler(&_forEachExceptionHandlerFn);
236 if (_exceptionHandled) {
237 if (resultCallback) {
238 auto result = SpecRunResult::failed(
239 self, spec, _currentExceptionMessage.c_str()
240 );
241 resultCallback->invoke(result.get());
242 } else {
243 _Log_("Code block error: {}", _currentExceptionMessage);
244 }
245 } else {
246 if (resultCallback) {
247 auto result =
248 SpecRunResult::failed(self, spec, "Unknown exception");
249 resultCallback->invoke(result.get());
250
251 } else {
252 _Log_("Code block error: Unhandled exception");
253 }
254 }
255 } else {
256 _Log_("No exception handlers registered");
257 }
258 } else {
259 _Log_("No currently running spec environment");
260 }
261 }
