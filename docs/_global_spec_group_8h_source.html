47 void component_defined(ISpecComponent* component) {
48 component->data()->merge(&_metaDataForNextComponent);
49 component->tags()->merge(&_tagsForNextComponent);
50
51 // Built-in support for "skip"
52 if (_metaDataForNextComponent.has(META_DATA_KEY_SKIP)) {
53 auto skip = _metaDataForNextComponent.get(META_DATA_KEY_SKIP)->bool_value();
54 component->mark_skipped(skip);
55 }
56
57 // Built-in support for "timeout"
58 if (_metaDataForNextComponent.has(META_DATA_KEY_TIMEOUT)) {
59 if (auto* hasCodeBlock = dynamic_cast<ISpecHasCodeBlock*>(component)) {
60 auto timeout =
61 _metaDataForNextComponent.get(META_DATA_KEY_TIMEOUT)->int_value();
62 hasCodeBlock->code_block()->set_timeout_ms(timeout);
63 }
64 }
65
66 _metaDataForNextComponent.clear();
67 _tagsForNextComponent.clear();
68 }
